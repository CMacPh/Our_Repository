pengle_decrease_sacrifices_10k = {
	add_to_variable = {
		var = current_sacrifices
		value = -10000
	}
}

convert_pops_pengle = {
	if = {
		limit = { has_war = no }
		every_owned_state = {
			limit = {
				NOT = { is_core_of = ROOT }
			}
			decrease_population_pengle = yes
		}
	}
	else = {
		every_state = {
			limit = {
				NOT = { is_core_of = ROOT }
				OR = {
					is_owned_by = ROOT
					is_controlled_by = ROOT
				}
			}
			decrease_population_pengle = yes
		}
	}
}
